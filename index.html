                const parseCSV = (text) => {

                const rows = text.split(/\r?\n/).slice(1);

                return rows.map((row, index) => {

                    const cols = row.split(',');

                    if (cols.length < 3) return null; 

                    

                    const name = cols[0]?.trim();

                    const status = cols[1]?.trim();

                    const progressString = cols[2]?.trim().replace('%', '');

                    const progress = parseInt(progressString) || 0;


                    if (!name) return null; 


                    return {

                    id: index + 1,

                    name: name,

                    status: status,

                    progress: progress

                    };

                }).filter(Boolean);
